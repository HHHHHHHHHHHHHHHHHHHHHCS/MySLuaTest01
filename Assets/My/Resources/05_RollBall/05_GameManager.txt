import "UnityEngine"

local gameManager = {}
local prefabsData
local uiManager,foodManager,player


function main()
    Init()
    return gameManager
end

function gameManager.Awake()

end

function gameManager.Start()

end

function gameManager.Update()
    for _,v in pairs(gameManager.updateList) do
        v()
    end
end

function Init()
    gameManager.updateList={}
    prefabsData=GameObject.Find("PrefabsData"):GetComponent("_05_Prefabs")

    GameObject.Instantiate(prefabsData.worldPrefab,parent)

    uiManager=dofile('05_RollBall/05_UIManager')
    foodManager=dofile('05_RollBall/05_FoodManager')
    player=dofile('05_RollBall/05_Player')

    player:Init(gameManager,prefabsData)
    foodManager:Init(gameManager,prefabsData)
    uiManager:Init(gameManager,prefabsData)
end
